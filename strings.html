9. Program to Check Anagram


import java.util.Arrays;
import java.util.Scanner;

public class AnagramChecker {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter first string: ");
        String str1 = scanner.nextLine();
        System.out.print("Enter second string: ");
        String str2 = scanner.nextLine();

        char[] arr1 = str1.replaceAll("\\s", "").toLowerCase().toCharArray();
        char[] arr2 = str2.replaceAll("\\s", "").toLowerCase().toCharArray();
        Arrays.sort(arr1);
        Arrays.sort(arr2);

        if (Arrays.equals(arr1, arr2)) {
            System.out.println("The strings are anagrams.");
        } else {
            System.out.println("The strings are not anagrams.");
        }
        
        scanner.close();
    }
}

15. Program to count the total number of punctuation characters exists in a String


import java.util.Scanner;

public class PunctuationCounter {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();
        int count = 0;

        for (char ch : input.toCharArray()) {
            if (isPunctuation(ch)) {
                count++;
            }
        }
        
        System.out.println("Total punctuation characters: " + count);
        scanner.close();
    }

    private static boolean isPunctuation(char ch) {
        return "!.,;:'\"?()[]{}".indexOf(ch) != -1;
    }
}

16. Program to count the total number of vowels and consonants in a string


import java.util.Scanner;

public class VowelConsonantCounter {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();
        int vowels = 0, consonants = 0;

        for (char ch : input.toLowerCase().toCharArray()) {
            if (Character.isLetter(ch)) {
                if ("aeiou".indexOf(ch) != -1)
                vowels++;
                } else {
                    consonants++;
                }
            }
        }

        System.out.println("Total vowels: " + vowels);
        System.out.println("Total consonants: " + consonants);
        
        scanner.close();
    }
}

19. Program to divide a string in 'N' equal parts


import java.util.Scanner;

public class DivideString {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();
        System.out.print("Enter the number of parts: ");
        int n = scanner.nextInt();

        if (n <= 0 || n > input.length()) {
            System.out.println("Invalid number of parts.");
            return;
        }

        int partLength = input.length() / n;
        for (int i = 0; i < n; i++) {
            System.out.println(input.substring(i * partLength, (i + 1) * partLength));
        }
        
        scanner.close();
    }
}

20. Program to find all subsets of a string


import java.util.Scanner;

public class StringSubsets {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();
        int n = input.length();
        int subsetCount = (1 << n); // 2^n subsets

        for (int i = 0; i < subsetCount; i++) {
            StringBuilder subset = new StringBuilder();
            for (int j = 0; j < n; j++) {
                if ((i & (1 << j)) > 0) {
                    subset.append(input.charAt(j));
                }
            }
            System.out.println(subset);
        }
        
        scanner.close();
    }
}

21. Program to find all the permutations of a string


import java.util.Scanner;

public class StringPermutations {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();
        permute(input, "");
        scanner.close();
    }

    private static void permute(String str, String result) {
        if (str.isEmpty()) {
            System.out.println(result);
        } else {
            for (int i = 0; i < str.length(); i++) {
                permute(str.substring(0, i) + str.substring(i + 1), result + str.charAt(i));
            }
        }
    }
}

22. Program to find maximum and minimum occurring character in a string


import java.util.HashMap;
import java.util.Map;
import java.util.Scanner;

public class MaxMinChar {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();

        Map charCount = new HashMap<>();
        for (char ch : input.toCharArray()) {
            charCount.put(ch, charCount.getOrDefault(ch, 0) + 1);
        }

        char maxChar = ' ';
        char minChar = ' ';
        int maxCount = 0;
        int minCount = Integer.MAX_VALUE;

        for (Map.Entry entry : charCount.entrySet()) {
            char currentChar = entry.getKey();
            int count = entry.getValue();

            if (count > maxCount) {
                maxCount = count;
                maxChar = currentChar;
            }

            if (count < minCount) {
                minCount = count;
                minChar = currentChar;
            }
        }

        System.out.println("Maximum occurring character: " + maxChar + " (" + maxCount + " times)");
        System.out.println("Minimum occurring character: " + minChar + " (" + minCount + " times)");

        scanner.close();
    }
}
